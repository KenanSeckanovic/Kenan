17  package com.acme.Krankenhaus.dev;
18  
19  import org.slf4j.LoggerFactory;
20  import org.springframework.boot.autoconfigure.condition.ConditionalOnCloudPlatform;
21  import org.springframework.boot.context.event.ApplicationReadyEvent;
22  import org.springframework.context.ApplicationListener;
23  import org.springframework.context.annotation.Bean;
24  import static org.springframework.boot.cloud.CloudPlatform.KUBERNETES;
25  import static org.springframework.context.annotation.Bean.Bootstrap.BACKGROUND;
26  
27  /// Protokoll-Ausgabe, wenn Kubernetes erkannt wird.
28  ///
29  /// @author [Jürgen Zimmermann](mailto:Juergen.Zimmermann@h-ka.de)
30  sealed interface K8s permits DevConfig {
31      /// Protokoll-Ausgabe, wenn Kubernetes erkannt wird.
32      ///
33      /// @return Listener zur Ausgabe, ob Kubernetes erkannt wird.
34      @Bean(bootstrap = BACKGROUND)
35      @ConditionalOnCloudPlatform(KUBERNETES)
36      default ApplicationListener<ApplicationReadyEvent> detectK8s() {
37          return _ -> LoggerFactory.getLogger(K8s.class).debug("Plattform \"Kubernetes\"");
38      }
39  }
