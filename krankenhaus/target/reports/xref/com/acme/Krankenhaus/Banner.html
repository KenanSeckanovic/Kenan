17  package com.acme.Krankenhaus;
18  
19  // import module java.base;
20  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
21  import java.net.InetAddress;
22  import java.net.UnknownHostException;
23  import java.util.Locale;
24  import java.util.Objects;
25  import org.springframework.boot.SpringBootVersion;
26  import org.springframework.core.SpringVersion;
27  
28  /// Banner als String-Konstante für den Start des Servers.
29  ///
30  /// @author [Jürgen Zimmermann](mailto:Juergen.Zimmermann@h-ka.de)
31  @SuppressWarnings({
32      "AccessOfSystemProperties",
33      "CallToSystemGetenv",
34      "UtilityClassCanBeEnum",
35      "UtilityClass",
36      "ClassUnconnectedToPackage"
37  })
38  @SuppressFBWarnings("NP_NULL_ON_SOME_PATH_FROM_RETURN_VALUE")
39  final class Banner {
40  
41      // http://patorjk.com/software/taag/#p=display&f=Slant&t=Krankenhaus%202024.10.1
42      private static final String FIGLET = """
43              __                   __        ___   ____ ___  __ __   _______   ___
44             / /____  ______  ____/ /__     |__ \\ / __ \\__ \\/ // /  <  / __ \\ <  /
45            / //_/ / / / __ \\/ __  / _ \\    __/ // / / /_/ / // /_  / / / / / / /
46           / ,< / /_/ / / / / /_/ /  __/   / __// /_/ / __/__  __/ / / /_/ / / /
47          /_/|_|\\__,_/_/ /_/\\__,_/\\___/   /____/\\____/____/ /_/ (_)_/\\____(_)_/
48          """;
49      private static final String SERVICE_HOST = System.getenv("Krankenhaus_SERVICE_HOST");
50      private static final String KUBERNETES = SERVICE_HOST == null
51          ? "N/A"
52          : "Krankenhaus_SERVICE_HOST=" + SERVICE_HOST + ", Krankenhaus_SERVICE_PORT=" + System.getenv("Krankenhaus_SERVICE_PORT");
53  
54      /// Banner für den Server-Start.
55      static final String TEXT = """
56  
57          $figlet
58          (C) Juergen Zimmermann, Hochschule Karlsruhe
59          Version             2024.10.1
60          Spring Boot         $springBoot
61          Spring Framework    $spring
62          Java                $java
63          Betriebssystem      $os
64          Rechnername         $rechnername
65          IP-Adresse          $ip
66          Heap: Size          $heapSize
67          Heap: Free          $heapFree
68          Kubernetes          $kubernetes
69          Username            $username
70          JVM Locale          $locale
71          OpenAPI             /swagger-ui.html /v3/api-docs.yaml
72          """
73          .replace("$figlet", FIGLET)
74          .replace("$springBoot", SpringBootVersion.getVersion())
75          .replace("$spring", Objects.requireNonNull(SpringVersion.getVersion()))
76          .replace("$java", Runtime.version() + " - " + System.getProperty("java.vendor"))
77          .replace("$os", System.getProperty("os.name"))
78          .replace("$rechnername", getLocalhost().getHostName())
79          .replace("$ip", getLocalhost().getHostAddress())
80          .replace("$heapSize", Runtime.getRuntime().totalMemory() / (1024L * 1024L) + " MiB")
81          .replace("$heapFree", Runtime.getRuntime().freeMemory() / (1024L * 1024L) + " MiB")
82          .replace("$kubernetes", KUBERNETES)
83          .replace("$username", System.getProperty("user.name"))
84          .replace("$locale", Locale.getDefault().toString());
85  
86      @SuppressWarnings("ImplicitCallToSuper")
87      private Banner() {
88      }
89  
90      private static InetAddress getLocalhost() {
91          try {
92              return InetAddress.getLocalHost();
93          } catch (final UnknownHostException ex) {
94              throw new IllegalStateException(ex);
95          }
96      }
97  }
