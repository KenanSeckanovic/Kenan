17  package com.acme.Krankenhaus.config;
18  
19  import org.springframework.aot.hint.RuntimeHints;
20  import org.springframework.aot.hint.RuntimeHintsRegistrar;
21  import org.springframework.context.annotation.ImportRuntimeHints;
22  
23  /**
24   * "RuntimeHints" für GraalVM, damit die PEM- und CRT-Dateien zur Laufzeit gefunden werden.
25   */
26  // https://stackoverflow.com/questions/...
27  // ...76287163/how-to-specify-the-location-of-a-keystore-file-with-spring-aot-processing
28  @ImportRuntimeHints(CertificateRuntimeHints.CertificateResourcesRegistrar.class)
29  public class CertificateRuntimeHints {
30  
31      /**
32       * Standardkonstruktor für die Klasse CertificateRuntimeHints.
33       * Wird automatisch erstellt, wenn keine zusätzlichen Initialisierungen benötigt werden.
34       */
35      public CertificateRuntimeHints() {
36          // Standardkonstruktor - keine Logik erforderlich
37      }
38  
39      /**
40       * Registrirung der PEM- und CRT-Dateien für GraalVM.
41       */
42      static class CertificateResourcesRegistrar implements RuntimeHintsRegistrar {
43  
44          /**
45           * Standardkonstruktor für die innere Klasse CertificateResourcesRegistrar.
46           * Initialisiert die Klasse für die Registrierung der Ressourcenmuster.
47           */
48          public CertificateResourcesRegistrar() {
49              // Standardkonstruktor - keine Logik erforderlich
50          }
51  
52          /**
53           * Registriert Muster für Ressourcen, damit PEM- und CRT-Dateien zur Laufzeit verfügbar sind.
54           *
55           * @param hints       das {@link RuntimeHints}-Objekt zur Registrierung der Ressourcenmuster
56           * @param classLoader der {@link ClassLoader}, der verwendet wird
57           */
58          @Override
59          public void registerHints(final RuntimeHints hints, final ClassLoader classLoader) {
60              hints.resources().registerPattern("*.pem");
61              hints.resources().registerPattern("*.crt");
62          }
63      }
64  
65  }
