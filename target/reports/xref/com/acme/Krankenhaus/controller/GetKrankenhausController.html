11  import org.springframework.web.bind.annotation.GetMapping;

15  
16  import java.util.List;
17  
18  import static org.springframework.http.MediaType.APPLICATION_JSON_VALUE;
19  
20  @RestController
21  @RequestMapping(GetKrankenhausController.API_PATH)
22  @OpenAPIDefinition(info = @Info(title = "Krankenhaus API", version = "v1"))
23  public class GetKrankenhausController {
24  
25      public static final String API_PATH = "/krankenhaus";
26      private static final Logger LOGGER = LoggerFactory.getLogger(GetKrankenhausController.class);
27  
28      private final KrankenhausReadService krankenhausReadService;
29  
30      public GetKrankenhausController(final KrankenhausReadService krankenhausReadService) {
31          this.krankenhausReadService = krankenhausReadService;
32      }
33  
34      @GetMapping(produces = APPLICATION_JSON_VALUE)
35      @Operation(summary = "Suche alle Krankenhäuser", tags = "Suchen")
36      @ApiResponse(responseCode = "200", description = "Alle Krankenhäuser erfolgreich gefunden")
37      @ApiResponse(responseCode = "404", description = "Keine Krankenhäuser in der Datenbank gefunden")
38      public List<Krankenhaus> getAll() {
39          LOGGER.info("Suche nach allen Krankenhäuser in der Datenbank gestartet");
40  
41          final List<Krankenhaus> krankenhaus = krankenhausReadService.getAll();
42          LOGGER.info("Suche nach allen Krankenhäuser in der Datenbank abgeschlossen");
43          return krankenhaus;
44      }
45  
46      @GetMapping(path = "/{id}", produces = APPLICATION_JSON_VALUE)
47      @Operation(summary = "Suche Krankenhaus nach id", tags = "Suchen")
48      @ApiResponse(responseCode = "200", description = "krankenhaus erfolgreich gefunden")
49      @ApiResponse(responseCode = "404", description = "Kein Krankenhaus mit dieser id gefunden")
50      public Krankenhaus getByID(@PathVariable final String id) {
51          LOGGER.info("Suche nach Krankenhaus mit id: {}", id);
52  
53          final Krankenhaus krankenhaus = krankenhausReadService.getByID(id);
54          LOGGER.info("Suche nach Krankenhaus mit id {} abgeschlossen", id);
55          return krankenhaus;
56      }
57  
58  }
