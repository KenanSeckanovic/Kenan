3   import com.acme.Krankenhaus.entity.Krankenhaus;
4   import org.slf4j.Logger;
5   import org.slf4j.LoggerFactory;
6   import org.springframework.lang.NonNull;
7   import org.springframework.stereotype.Repository;
8   import java.util.List;
9   import java.util.Map;
10  import java.util.Objects;
11  import java.util.Optional;
12  import static com.acme.Krankenhaus.repository.KrankenhausDB.KRANKENHAUSER;
13  import static java.util.Collections.emptyList;
14  
15  /**
16   * Repository-Klasse für den Zugriff auf {@link Krankenhaus}-Daten.
17   * Diese Klasse stellt Methoden zum Abrufen von Krankenhausdaten aus einer statischen Datenquelle bereit.
18   */
19  @Repository
20  public class KrankenhausRepository {
21      private static final Logger LOGGER = LoggerFactory.getLogger(KrankenhausRepository.class);
22  
23      /**
24       * Konstruktor für das KrankenhausRepository.
25       * Aktuell keine speziellen Initialisierungen erforderlich.
26       */
27      public KrankenhausRepository() {
28          // Kein spezieller Initialisierungsbedarf
29      }
30  
31      /**
32       * Ruft alle Krankenhäuser ab.
33       *
34       * @return eine Liste aller verfügbaren {@link Krankenhaus}-Objekte
35       */
36      public @NonNull List<Krankenhaus> findAll() {
37          return KRANKENHAUSER;
38      }
39  
40      /**
41       * Sucht ein Krankenhaus anhand der angegebenen ID.
42       *
43       * @param id die ID des gesuchten Krankenhauses
44       * @return ein {@link Optional} mit dem gefundenen Krankenhaus,
45       *         oder ein leeres Optional, wenn kein Krankenhaus mit der angegebenen ID gefunden wurde
46       */
47      public @NonNull Optional<Krankenhaus> findByID(@NonNull final String id) {
48          LOGGER.debug("Suche nach Krankenhaus mit id: {}", id);
49  
50          Optional<Krankenhaus> krankenhaus = KRANKENHAUSER.stream()
51              .filter(a -> Objects.equals(id, a.getUuid().toString()))
52              .findFirst();
53  
54          LOGGER.debug("Ergebnis der Suche nach id {}: {}", id, krankenhaus);
55          return krankenhaus;
56      }
57  
58      /**
59       * Sucht nach Krankenhäusern basierend auf den angegebenen Suchkriterien.
60       * Unterstützte Kriterien: "standort" und "name".
61       *
62       * @param suchkriterien eine Map von Suchkriterien zur Filterung
63       * @return eine Liste der gefundenen Krankenhäuser oder eine leere Liste, wenn keine Übereinstimmungen vorhanden sind
64       */
65      public @NonNull List<Krankenhaus> find(final Map<String, ? extends List<String>> suchkriterien) {
66          LOGGER.debug("find: suchkriterien={}", suchkriterien);
67  
68          if (suchkriterien.isEmpty()) {
69              return findAll();
70          }
71  
72          for (final var entry : suchkriterien.entrySet()) {
73              switch (entry.getKey()) {
74                  case "standort" -> {
75                      return findByStandort(entry.getValue().getFirst());
76                  }
77                  case "name" -> {
78                      return findByName(entry.getValue().getFirst());
79                  }
80                  default -> {
81                      LOGGER.debug("find: ungueltiges Suchkriterium={}", entry.getKey());
82                      return emptyList();
83                  }
84              }
85          }
86  
87          return emptyList();
88      }
89  
90      /**
91       * Sucht nach Krankenhäusern, deren Namen das angegebene Schlüsselwort enthalten.
92       *
93       * @param name der Name oder ein Teil davon, nach dem gesucht wird
94       * @return eine Liste der gefundenen {@link Krankenhaus}-Objekte mit übereinstimmendem Namen
95       */
96      public @NonNull List<Krankenhaus> findByName(final String name) {
97          LOGGER.debug("findByName: name={}", name);
98          final var krankenhaeuser = KRANKENHAUSER.stream()
99              .filter(krankenhaus -> krankenhaus.getName().contains(name))
100             .toList();
101         LOGGER.debug("findByName: krankenhaeuser={}", krankenhaeuser);
102         return krankenhaeuser;
103     }
104 
105     /**
106      * Sucht nach Krankenhäusern, die sich im angegebenen Standort befinden.
107      *
108      * @param standort der Standort (Stadt) des Krankenhauses
109      * @return eine Liste der gefundenen {@link Krankenhaus}-Objekte mit übereinstimmendem Standort
110      */
111     public @NonNull List<Krankenhaus> findByStandort(final String standort) {
112         LOGGER.debug("findByStandort: standort={}", standort);
113         final var krankenhaeuser = KRANKENHAUSER.stream()
114             .filter(krankenhaus -> krankenhaus.getLageort().getStadt().contains(standort))
115             .toList();
116         LOGGER.debug("findByStandort: krankenhaeuser={}", krankenhaeuser);
117         return krankenhaeuser;
118     }
119 }
