14  import static com.acme.Krankenhaus.repository.KrankenhausDB.KRANKENHAUSER;
15  
16  /**
17   * Repository-Klasse für den Zugriff auf {@link Krankenhaus}-Daten.
18   * Diese Klasse stellt Methoden zum Abrufen von Krankenhausdaten aus einer statischen Datenquelle bereit.
19   */
20  @Repository
21  public class KrankenhausRepository {
22  
23      private static final Logger LOGGER = LoggerFactory.getLogger(KrankenhausRepository.class);
24  
25      /**
26       * Konstruktor für das KrankenhausRepository.
27       * Aktuell keine speziellen Initialisierungen erforderlich.
28       */
29      public KrankenhausRepository() {
30          // Kein spezieller Initialisierungsbedarf
31      }
32  
33      /**
34       * Ruft alle Krankenhäuser ab.
35       *
36       * @return eine Liste aller verfügbaren {@link Krankenhaus}-Objekte
37       */
38      public @NonNull List<Krankenhaus> getAll() {
39          return KRANKENHAUSER;
40      }
41  
42      /**
43       * Sucht ein Krankenhaus anhand der angegebenen ID.
44       *
45       * @param id die ID des gesuchten Krankenhauses
46       * @return ein {@link Optional} mit dem gefundenen Krankenhaus,
47       *         oder ein leeres Optional, wenn kein Krankenhaus mit der angegebenen ID gefunden wurde
48       */
49      public @NonNull Optional<Krankenhaus> getByID(@NonNull final String id) {
50          LOGGER.debug("Suche nach Krankenhaus mit id: {}", id);
51  
52          Optional<Krankenhaus> krankenhaus = KRANKENHAUSER.stream()
53              .filter(a -> Objects.equals(id, a.getUuid().toString()))
54              .findFirst();
55  
56          LOGGER.debug("Ergebnis der Suche nach id {}: {}", id, krankenhaus);
57          return krankenhaus;
58      }
59  }
