1   package com.acme.Krankenhaus.service;
2   
3   import com.acme.Krankenhaus.entity.Krankenhaus;
4   import com.acme.Krankenhaus.repository.KrankenhausRepository;
5   import org.slf4j.Logger;
6   import org.slf4j.LoggerFactory;
7   import org.springframework.lang.NonNull;
8   import org.springframework.stereotype.Service;
9   
10  import java.util.List;
11  
12  @Service
13  public class KrankenhausReadService {
14  
15      private static final Logger LOGGER = LoggerFactory.getLogger(KrankenhausReadService.class);
16      private final KrankenhausRepository krankenhausRepository;
17  
18      public KrankenhausReadService( final KrankenhausRepository krankenhausRepository) {
19          this.krankenhausRepository = krankenhausRepository;
20      }
21  
22      public @NonNull List<Krankenhaus> getAll() {
23          LOGGER.debug("Starte Abruf aller Krankenhaus");
24  
25          List<Krankenhaus> krankenhauser = krankenhausRepository.getAll();
26          if (krankenhauser.isEmpty()) {
27              throw new IllegalArgumentException("Keine Autohäuser in der Datenbank gefunden.");
28          }
29  
30          return krankenhauser;
31      }
32  
33      public @NonNull Krankenhaus getByID(String id) {
34          LOGGER.debug("Starte Suche nach Krankenhaus mit id: {}", id);
35  
36          Krankenhaus krankenhaus = krankenhausRepository.getByID(id)
37              .orElseThrow(() -> new IllegalArgumentException("Kein Krankenhaus für die angegebene id gefunden."));
38  
39          LOGGER.debug("Suche nach Krankenhaus mit id beendet");
40          return krankenhaus;
41      }
42  }
