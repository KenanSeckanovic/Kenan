1   package com.acme.Krankenhaus.controller;
2   
3   import com.acme.Krankenhaus.entity.Krankenhaus;
4   import com.acme.Krankenhaus.service.KrankenhausReadService;
5   import io.swagger.v3.oas.annotations.OpenAPIDefinition;
6   import io.swagger.v3.oas.annotations.Operation;
7   import io.swagger.v3.oas.annotations.info.Info;
8   import io.swagger.v3.oas.annotations.responses.ApiResponse;
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  import org.springframework.web.bind.annotation.GetMapping;
12  import org.springframework.web.bind.annotation.PathVariable;
13  import org.springframework.web.bind.annotation.RequestMapping;
14  import org.springframework.web.bind.annotation.RestController;
15  
16  import java.util.List;
17  
18  import static org.springframework.http.MediaType.APPLICATION_JSON_VALUE;
19  
20  /**
21   * REST-Controller, der Endpunkte zum Abrufen von Krankenhausdaten bereitstellt.
22   */
23  @RestController
24  @RequestMapping(GetKrankenhausController.API_PATH)
25  @OpenAPIDefinition(info = @Info(title = "Krankenhaus API", version = "v1"))
26  public class GetKrankenhausController {
27  
28      /** Der Pfad für die Krankenhaus-API. */
29      public static final String API_PATH = "/krankenhaus";
30  
31      private static final Logger LOGGER = LoggerFactory.getLogger(GetKrankenhausController.class);
32  
33      private final KrankenhausReadService krankenhausReadService;
34  
35      /**
36       * Konstruktor für die Initialisierung des Controllers mit dem erforderlichen Service.
37       *
38       * @param krankenhausReadService der Service, der Krankenhausdaten bereitstellt
39       */
40      public GetKrankenhausController(final KrankenhausReadService krankenhausReadService) {
41          this.krankenhausReadService = krankenhausReadService;
42      }
43  
44      /**
45       * Endpunkt zum Abrufen aller Krankenhäuser in der Datenbank.
46       *
47       * @return eine Liste aller Krankenhäuser
48       */
49      @GetMapping(produces = APPLICATION_JSON_VALUE)
50      @Operation(summary = "Suche alle Krankenhäuser", tags = "Suchen")
51      @ApiResponse(responseCode = "200", description = "Alle Krankenhäuser erfolgreich gefunden")
52      @ApiResponse(responseCode = "404", description = "Keine Krankenhäuser in der Datenbank gefunden")
53      public List<Krankenhaus> getAll() {
54          LOGGER.info("Suche nach allen Krankenhäuser in der Datenbank gestartet");
55  
56          final List<Krankenhaus> krankenhaus = krankenhausReadService.getAll();
57          LOGGER.info("Suche nach allen Krankenhäuser in der Datenbank abgeschlossen");
58          return krankenhaus;
59      }
60  
61      /**
62       * Endpunkt zum Abrufen eines Krankenhauses anhand der ID.
63       *
64       * @param id die ID des gesuchten Krankenhauses
65       * @return das Krankenhaus mit der angegebenen ID, falls vorhanden
66       */
67      @GetMapping(path = "/{id}", produces = APPLICATION_JSON_VALUE)
68      @Operation(summary = "Suche Krankenhaus nach id", tags = "Suchen")
69      @ApiResponse(responseCode = "200", description = "krankenhaus erfolgreich gefunden")
70      @ApiResponse(responseCode = "404", description = "Kein Krankenhaus mit dieser id gefunden")
71      public Krankenhaus getByID(@PathVariable final String id) {
72          LOGGER.info("Suche nach Krankenhaus mit id: {}", id);
73  
74          final Krankenhaus krankenhaus = krankenhausReadService.getByID(id);
75          LOGGER.info("Suche nach Krankenhaus mit id {} abgeschlossen", id);
76          return krankenhaus;
77      }
78  
79  }
